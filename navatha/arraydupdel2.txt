#include <stdio.h>

int main()
{
int i,j,n=5,k;
int a[5]={2,3,5,3,5};
{
    for(i=0;i<=n-1;i++)
    {
        for(j=i+1;j<=n-1;j++)
        {
            if(a[i]==a[j])
            {
            for(k=j;k<=n-1;k++)
            {
                a[k]=a[k+1];
            }
            j--;
            n--;
            }
            
        }   
       
        printf("%d",a[i]); 
    }
   
}
    return 0;
}
